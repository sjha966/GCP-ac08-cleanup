resource "null_resource" "detach_service_project" {
  provisioner "local-exec" {
    command = "gcloud compute shared-vpc associated-projects remove --host-project=${var.host_project} ${var.service_project}"
  }
  }

  resource "null_resource" "delete_shared_vpc_subnet" {
  provisioner "local-exec" {
    command = "gcloud compute networks subnets delete ${var.subnet_name} --region=${var.region}"
  }
  }
